#version 430 core

struct Camera
{
	vec3 worldCameraPosition;
	mat4 perspectiveMatrix;
	mat4 viewMatrix;
};
uniform Camera CurrentCamera;


layout (location = 0) in vec3 vertex_position;
layout (location = 0) out vec4 vertex_position_out;

void main(void)
{
	gl_Position = CurrentCamera.perspectiveMatrix * CurrentCamera.viewMatrix * vec4(vertex_position.x, vertex_position.y, vertex_position.z, 1.0);
	vertex_position_out = (vec4(vertex_position, 1.0));
}
