#version 430 core

in vec4 texture_coordinates;


layout(binding = 0) uniform sampler2D TextureSampler0;
layout(binding = 1) uniform sampler2D TextureSampler1;

layout(location = 0) uniform float weight_one;
layout(location = 1) uniform float weight_two;


out vec4 color;

void main()
{
	vec4 colorOne = texture(TextureSampler0, vec2(texture_coordinates.x, texture_coordinates.y));
	vec4 colorTwo = texture(TextureSampler1, vec2(texture_coordinates.x, texture_coordinates.y));

	color = vec4(colorOne.xyz * weight_one + colorTwo.xyz * weight_two, 1.0);
}
