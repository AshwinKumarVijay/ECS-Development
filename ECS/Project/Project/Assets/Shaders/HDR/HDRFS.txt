#version 430 core

//	TEXTURE SAMPLER
layout (location = 0) uniform sampler2D u_Texture;

//	TEXTURE COORDINATES.
layout (location = 0) in vec2 v_TextureCoordinates;

//	COLOR OUTPUT
layout (location = 0) out vec4 o_fragColor;

//	THE FRAGMENT SHADER MAIN.
void main(void)
{
	//	Sample the new fragment color.
	vec3 newFragmentColor = texture(u_Texture, v_TextureCoordinates).xyz;

	//	Reinhard Tone-Mapping.
	newFragmentColor = newFragmentColor / (newFragmentColor + vec3(1.0));

	//	Output the new fragment color.
	o_fragColor = vec4(newFragmentColor, 1.0);
}
