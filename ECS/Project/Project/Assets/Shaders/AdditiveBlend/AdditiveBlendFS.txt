#version 430 core

//	TEXTURE SAMPLER
layout (location = 0) uniform sampler2D u_TextureOne;
layout (location = 1) uniform sampler2D u_TextureTwo;

layout (location = 2) uniform float u_WeightOne = 1.0;
layout (location = 3) uniform float u_WeightTwo = 1.0;

//	TEXTURE COORDINATES.
layout (location = 0) in vec2 v_TextureCoordinates;

//	COLOR OUTPUT
layout (location = 0) out vec4 o_fragColor;

//	THE FRAGMENT SHADER MAIN.
void main(void)
{

	vec4 textureColorOne = texture(u_TextureOne, v_TextureCoordinates);
	vec4 textureColorTwo = texture(u_TextureTwo, v_TextureCoordinates);
	
	//	Sample the new fragment color.
	o_fragColor = vec4(u_WeightOne * textureColorOne + u_WeightTwo * textureColorTwo);
}
